create table raw_material (
  id int AUTO_INCREMENT,
  name varchar(20),
  supplier varchar(30),
  description varchar(50),
  price decimal(8,3),
  PRIMARY KEY(id)
);


create table component_material (
  id int AUTO_INCREMENT,
  name varchar(20),
  supplier varchar(30),
  description varchar(50),
  price decimal(8,3),
  PRIMARY KEY(id)
);


create table finished_Product (
  id int AUTO_INCREMENT,
  name varchar(20),
  description varchar(50),
  price decimal(8,3),
  PRIMARY KEY(id)
);

create table purchase_details (
orderId integer AUTO_INCREMENT,
rawQuantity long,
componentQuantity long,
totalPrice decimal(10,3),
Primary Key (orderId)
);

create table sales_details (
saleId integer AUTO_INCREMENT,
finishedQuantity long,
totalPrice decimal(10,3),
Primary Key (saleId)
);
